```sql
SELECT ITEM_ID, ITEM_NAME, RARITY
FROM ITEM_INFO JOIN ITEM_TREE USING(ITEM_ID)
WHERE PARENT_ITEM_ID IN (
    SELECT ITEM_ID AS ID
    FROM ITEM_INFO INF JOIN ITEM_TREE TR USING(ITEM_ID)
    WHERE INF.RARITY = 'RARE' 
    )
ORDER BY 1 DESC
```
