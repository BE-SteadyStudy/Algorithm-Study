class Solution {
    static boolean[] visit;
    static int[][] arr;
    public int solution(int n, int[][] wires) {
        int answer = 101;
        arr = new int[n+1][n+1];

        for(int[] wire : wires) {
            int x = wire[0];
            int y = wire[1];
            arr[x][y] = arr[y][x] = 1;
        }
        int diff = 0;
        for(int[] wire : wires) {
            int x = wire[0];
            int y = wire[1];
            visit = new boolean[n+1];
            arr[x][y] = arr[y][x] = 0;
            diff = 0;
            for(int i=1;i<=n;i++) {
                if(!visit[i]) {
                    int count = bfs(i, n);
                    diff = Math.abs(diff-count);
                }
            }
            if(answer > diff) {
                answer = diff;
            }
            arr[x][y] = arr[y][x] = 1;
        }
        return answer;
    }

    private static int bfs(int start, int n) {
        Queue<Integer> q = new LinkedList<>();
        q.offer(start);
        int count = 1;
        while(!q.isEmpty()) {
            int next = q.poll();
            visit[next] = true;
            for(int i=1;i<=n;i++) {
                if(arr[next][i]==1 && !visit[i]) {
                    q.add(i);
                    visit[i] = true;
                    count++;
                }
            }
        }
        return count;
    }
}
