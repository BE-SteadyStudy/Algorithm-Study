class Solution {
    public int[] solution(int brown, int yellow) {

        int answer[] = new int[2];

        int sum = brown + yellow;

        for(int i=3; i<sum; i++) {
            int j = sum / i;

            if((sum % i != 0) || (j < 3)) continue;

            if(((j-2)*(i-2)) == yellow) {
                answer[0] = j;
                answer[1] = i;
                return answer;
            }
        }

        return answer;
    }
}
